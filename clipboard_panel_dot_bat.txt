function New-TemplateItem {
    param(
        [string]$Id,
        [string]$Title,
        [string]$Template
    )

    if ([string]::IsNullOrWhiteSpace($Id)) {
        $Id = [guid]::NewGuid().ToString()
    }

    [pscustomobject]@{
        id       = [string]$Id
        title    = [string]$Title
        template = [string]$Template
    }
}

function Get-ObjectPropertyValue {
    param(
        [object]$Object,
        [string]$Name,
        [object]$DefaultValue = $null
    )

    if ($null -eq $Object) {
        return $DefaultValue
    }

    $property = $Object.PSObject.Properties[$Name]
    if ($null -eq $property) {
        return $DefaultValue
    }

    return $property.Value
}

function Load-Templates {
    if (-not (Test-Path -LiteralPath $script:DataFile)) {
        return @()
    }

    try {
        $json = [System.IO.File]::ReadAllText($script:DataFile, [System.Text.Encoding]::UTF8)
        if ([string]::IsNullOrWhiteSpace($json)) {
            return @()
        }

        $raw = $json | ConvertFrom-Json
        $rawItems = @()

        if ($raw -is [System.Array]) {
            $rawItems = $raw
        }
        elseif ($null -ne $raw -and $raw.PSObject.Properties['templates']) {
            $rawItems = @($raw.templates)
        }

        $loadedItems = New-Object System.Collections.Generic.List[object]
        foreach ($item in @($rawItems)) {
            if ($null -eq $item) {
                continue
            }

            $title = [string](Get-ObjectPropertyValue $item 'title' '')
            $template = [string](Get-ObjectPropertyValue $item 'template' '')
            $id = [string](Get-ObjectPropertyValue $item 'id' '')

            $title = $title.Trim()
            if (-not [string]::IsNullOrWhiteSpace($title) -and -not [string]::IsNullOrWhiteSpace($template)) {
                $loadedItems.Add((New-TemplateItem -Id $id -Title $title -Template $template))
            }
        }

        return @($loadedItems.ToArray())
    }
    catch {
        Show-ErrorMessage "Could not read templates.json:`r`n$($_.Exception.Message)" 'JSON read error'
        return @()
    }
}

function Save-Templates {
    try {
        $items = @(
            foreach ($item in @($script:Templates)) {
                [ordered]@{
                    id       = [string]$item.id
                    title    = [string]$item.title
                    template = [string]$item.template
                }
            }
        )

        $data = [ordered]@{
            templates = $items
        }

        $json = $data | ConvertTo-Json -Depth 20
        $utf8NoBom = New-Object -TypeName System.Text.UTF8Encoding -ArgumentList $false
        [System.IO.File]::WriteAllText($script:DataFile, $json, $utf8NoBom)
    }
    catch {
        Show-ErrorMessage "Could not save templates.json:`r`n$($_.Exception.Message)" 'JSON write error'
    }
}

function Get-GridColumns {
    if ($null -eq $script:ScrollPanel) {
        return $script:DefaultGridColumns
    }

    $width = $script:ScrollPanel.ClientSize.Width
    if ($width -le 0) {
        $width = $script:MainForm.ClientSize.Width
    }

    $cellWidth = $script:ButtonSize + $script:CellGap
    $columns = [int][Math]::Floor($width / $cellWidth)
    if ($columns -lt 1) {
        $columns = 1
    }
    if ($columns -gt $script:MaxGridColumns) {
        $columns = $script:MaxGridColumns
    }
    return $columns
}

function New-ScaledImage {
    param(
        [string]$Path,
        [int]$Width,
        [int]$Height
    )

    if (-not (Test-Path -LiteralPath $Path)) {
        return $null
    }

    try {
        $source = [System.Drawing.Image]::FromFile($Path)
        try {
            $bitmap = New-Object -TypeName System.Drawing.Bitmap -ArgumentList $Width, $Height
            $graphics = [System.Drawing.Graphics]::FromImage($bitmap)
            try {
                $graphics.InterpolationMode = [System.Drawing.Drawing2D.InterpolationMode]::HighQualityBicubic
                $graphics.SmoothingMode = [System.Drawing.Drawing2D.SmoothingMode]::HighQuality
                $graphics.PixelOffsetMode = [System.Drawing.Drawing2D.PixelOffsetMode]::HighQuality
                $graphics.Clear([System.Drawing.Color]::Transparent)
                $graphics.DrawImage($source, 0, 0, $Width, $Height)
            }
            finally {
                $graphics.Dispose()
            }
            return $bitmap
        }
        finally {
            $source.Dispose()
        }
    }
    catch {
        return $null
    }
}

function Handle-TemplateClick {
    param([object]$Item)

    $now = Get-Date
    $elapsedMs = ($now - $script:LastClickTime).TotalMilliseconds
    $isDoubleClick = (
        $script:LastClickTemplateId -eq $Item.id -and
        $elapsedMs -le ($script:SingleClickDelayMs * 1.5)
    )

    if ($isDoubleClick) {
        if ($null -ne $script:ClickTimer) {
            $script:ClickTimer.Stop()
        }
        $script:PendingClickItem = $null
        $script:LastClickTemplateId = $null
        $script:LastClickTime = [DateTime]::MinValue
        Open-EditDialog $Item
        return
    }

    if ($null -ne $script:ClickTimer) {
        $script:ClickTimer.Stop()
    }

    $script:LastClickTemplateId = $Item.id
    $script:LastClickTime = $now
    $script:PendingClickItem = $Item
    $script:ClickTimer.Start()
}

function Render-Grid {
    if ($null -eq $script:GridTable) {
        return
    }

    $columns = Get-GridColumns
    $script:CurrentGridColumns = $columns

    $script:GridTable.SuspendLayout()
    try {
        $script:GridTable.Controls.Clear()
        $script:GridTable.ColumnStyles.Clear()
        $script:GridTable.RowStyles.Clear()

        $totalItems = @($script:Templates).Count + 1
        $rows = [int][Math]::Ceiling($totalItems / [double]$columns)
        if ($rows -lt 1) {
            $rows = 1
        }

        $script:GridTable.ColumnCount = $columns
        $script:GridTable.RowCount = $rows
        $script:GridTable.Width = $columns * ($script:ButtonSize + $script:CellGap)
        $script:GridTable.Height = $rows * ($script:ButtonSize + $script:CellGap)

        for ($col = 0; $col -lt $columns; $col++) {
            $script:GridTable.ColumnStyles.Add(
                (New-Object -TypeName System.Windows.Forms.ColumnStyle -ArgumentList ([System.Windows.Forms.SizeType]::Absolute), ($script:ButtonSize + $script:CellGap))
            ) | Out-Null
        }

        for ($row = 0; $row -lt $rows; $row++) {
            $script:GridTable.RowStyles.Add(
                (New-Object -TypeName System.Windows.Forms.RowStyle -ArgumentList ([System.Windows.Forms.SizeType]::Absolute), ($script:ButtonSize + $script:CellGap))
            ) | Out-Null
        }

        $index = 0
        foreach ($item in @($script:Templates)) {
            $row = [int][Math]::Floor($index / $columns)
            $col = $index % $columns
            New-TemplateButton -Item $item -Row $row -Column $col
            $index++
        }

        $addRow = [int][Math]::Floor($index / $columns)
        $addCol = $index % $columns
        New-AddButton -Row $addRow -Column $addCol
    }
    finally {
        $script:GridTable.ResumeLayout($true)
    }
}

function Add-Template {
    param([string]$Title, [string]$Template)

    $script:Templates = @($script:Templates) + (New-TemplateItem -Id ([guid]::NewGuid().ToString()) -Title $Title -Template $Template)
    Save-Templates
    Render-Grid
    Set-Status "Added template: $Title"
}

function Update-Template {
    param(
        [string]$TemplateId,
        [string]$Title,
        [string]$Template
    )

    foreach ($item in @($script:Templates)) {
        if ($item.id -eq $TemplateId) {
            $item.title = $Title
            $item.template = $Template
            break
        }
    }

    Save-Templates
    Render-Grid
    Set-Status "Updated template: $Title"
}

function Delete-Template {
    param([string]$TemplateId)

    $before = @($script:Templates).Count
    $script:Templates = @($script:Templates | Where-Object { $_.id -ne $TemplateId })
    $after = @($script:Templates).Count

    if ($after -ne $before) {
        Save-Templates
        Render-Grid
        Set-Status 'Deleted template'
    }
}

function Open-AddDialog {
    Show-TemplateDialog `
        -DialogTitle 'Add Template' `
        -InitialTitle '' `
        -InitialTemplate '' `
        -ShowDelete $false `
        -OnSave { param($title, $template) Add-Template -Title $title -Template $template }
}

function Open-EditDialog {
    param([object]$Item)

    Show-TemplateDialog `
        -DialogTitle "Edit Template: $($Item.title)" `
        -InitialTitle ([string]$Item.title) `
        -InitialTemplate ([string]$Item.template) `
        -ShowDelete $true `
        -OnSave { param($title, $template) Update-Template -TemplateId ([string]$Item.id) -Title $title -Template $template } `
        -OnDelete { Delete-Template -TemplateId ([string]$Item.id) }
}

function Reload-FromDisk {
    $script:Templates = @(Load-Templates)
    Render-Grid
    Set-Status 'Reloaded templates.json'
}

    $script:ScrollPanel.Controls.Add($script:GridTable)

    $script:StatusLabel = New-Object System.Windows.Forms.Label
    $script:StatusLabel.Dock = [System.Windows.Forms.DockStyle]::Fill
    $script:StatusLabel.ForeColor = [System.Drawing.ColorTranslator]::FromHtml('#2f6f2f')
    $script:StatusLabel.TextAlign = [System.Drawing.ContentAlignment]::MiddleLeft
    $root.Controls.Add($script:StatusLabel, 0, 3)

    $script:ResizeTimer = New-Object System.Windows.Forms.Timer
    $script:ResizeTimer.Interval = 80
    $script:ResizeTimer.Add_Tick({
        $script:ResizeTimer.Stop()
        $newColumns = Get-GridColumns
        if ($newColumns -ne $script:CurrentGridColumns) {
            Render-Grid
        }
    })

    $script:ScrollPanel.Add_SizeChanged({
        if ($null -ne $script:ResizeTimer) {
            $script:ResizeTimer.Stop()
            $script:ResizeTimer.Start()
        }
    })

    $script:ClickTimer = New-Object System.Windows.Forms.Timer
    $script:ClickTimer.Interval = $script:SingleClickDelayMs
    $script:ClickTimer.Add_Tick({
        $script:ClickTimer.Stop()
        if ($null -ne $script:PendingClickItem) {
            $itemToCopy = $script:PendingClickItem
            $script:PendingClickItem = $null
            $script:LastClickTemplateId = $null
            $script:LastClickTime = [DateTime]::MinValue
            Copy-Template $itemToCopy
        }
    })

    $script:Templates = @(Load-Templates)
    $script:AddButtonImage = New-ScaledImage -Path $script:AddIconFile -Width 42 -Height 42
    Set-Status "Data file: $script:DataFile"
    Render-Grid
}

try {
    Build-MainForm
    [System.Windows.Forms.Application]::Run($script:MainForm)
}
catch {
    Show-ErrorMessage "Clipboard Panel failed:`r`n$($_.Exception.Message)" 'Application error'
    exit 1
}
